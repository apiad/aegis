PRIMING = (
    "You are running inside aegis, a meta-harness. An MCP server named "
    "'aegis' is attached and (strict config) is your only MCP server. "
    "Your aegis handle is '{handle}'. Call its aegis_meta tool first to "
    "learn this environment and the aegis tools available to you, then "
    "proceed. When handing off, pass your handle '{handle}' as "
    "from_handle."

# BRIEFING: keep slice-1 text, then append a tools section naming
# aegis_list_sessions / aegis_list_agents / aegis_handoff, stating the
# agent's own handle is in its system prompt and is passed as from_handle,
# and that handoff is a one-way context transfer to a live peer.

def build_server(bridge):
    server = FastMCP("aegis")
    server.tool(aegis_meta)

    @server.tool
    async def aegis_list_sessions() -> list[dict]:
        """Live aegis sessions (peers you can hand off to)."""
        return [s.__dict__ for s in bridge.list_sessions()]

    @server.tool
    async def aegis_list_agents() -> list[str]:
        """Configured agent profiles that can be spawned."""
        return list(bridge.list_agents())

    @server.tool
    async def aegis_handoff(from_handle: str, target_handle: str,
                            context: str) -> str:
        """One-way context transfer to a live peer session.
        from_handle is your own aegis handle (see your system prompt)."""
        return await bridge.handoff(from_handle, target_handle, context)

    return server

(SessionInfo is a frozen dataclass; s.__dict__ works for @dataclass. If frozen=True blocks __dict__ on the runtime, use dataclasses.asdict(s) — import dataclasses; pick whichever the test proves and report.)

    # --- AppBridge ---
    def list_sessions(self):
        from aegis.mcp.bridge import SessionInfo
        active = self._active
        return [
            SessionInfo(handle=p.handle, agent_slug=p.agent_slug,
                        state=p.state.value, active=(p is active),
                        unseen=p.unseen)
            for p in self._panes
        ]

    def list_agents(self):
        return sorted(self._agents)

    async def handoff(self, from_handle: str, target_handle: str,
                      context: str) -> str:
        if from_handle == target_handle:
            return "handoff rejected: cannot hand off to yourself"
        target = next((p for p in self._panes
                       if p.handle == target_handle), None)
        if target is None:
            return (f"handoff rejected: no session {target_handle!r} "
                    f"(use aegis_list_sessions)")
        if target.state is AgentState.working:
            return (f"handoff rejected: {target_handle!r} is busy, "
                    f"retry shortly")
        await target.deliver_handoff(from_handle, context)
        return f"delivered to {target_handle}"

@pytest.mark.asyncio
async def test_live_agent_lists_sessions():
    # one-pane bridge stub: the agent should be able to call
    # aegis_list_sessions and get a result (its own handle).
    from aegis.mcp import AegisMCP, mcp_config_json
    import asyncio, json

    class OneSession:
        def list_sessions(self):
            from aegis.mcp.bridge import SessionInfo
            return [SessionInfo("lucid-knuth", "default",
                                "ready", True, False)]
        def list_agents(self): return ["default"]
        async def handoff(self, a, b, c): return "delivered to " + b

    mcp = AegisMCP()
    mcp.bind(OneSession())
    await mcp.start()
    try:
        argv = ["claude", "-p", "--input-format", "stream-json",
                "--output-format", "stream-json", "--replay-user-messages",
                "--verbose", "--permission-mode", "bypassPermissions",
                "--strict-mcp-config", "--mcp-config",
                mcp_config_json(mcp.url)]
        proc = await asyncio.create_subprocess_exec(
            *argv, stdin=asyncio.subprocess.PIPE,
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE)
        proc.stdin.write((json.dumps({"type": "user", "message":
            {"role": "user", "content":
             "Call aegis_list_sessions and report the handles."}})
            + "\n").encode())
        await proc.stdin.drain()
        out = (await asyncio.wait_for(
            proc.stdout.read(), timeout=150)).decode("utf-8", "replace")
        proc.kill()
        assert "aegis_list_sessions" in out
        assert "lucid-knuth" in out
    finally:
        await mcp.stop()
